    beta1 ~ dnorm(0, 0.1)
    alpha1 ~ dnorm(0, 0.368)
    beta0 ~ dnorm(0, 0.1)
    alpha0 ~ dnorm(0, 0.368)
    
    for(g in 1:G){   # g is the pixel index, there are G total pixels
    lambda[g] <- exp(beta1 * x[g] + beta0)
    logit(p.po[g]) <- w[g] * alpha1 + alpha0
    y.po[g] ~ dpois(lambda[g] * p.po[g])
    }
